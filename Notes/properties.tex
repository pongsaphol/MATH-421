\chapter{Properties of Real Number}

\begin{definition}
Given any $a\in\RR$, we define its absolute value to be
\[ |a| =
  \begin{cases}
    a       & \quad \text{if } a \geq 0 \\
    a  & \quad \text{if } a < 0
  \end{cases}
\]
\end{definition}

\begin{theorem}[Triangular Inequality]
Given $a, b\in\RR$, there holds
$$|a+b|\leq |a| + |b|$$
\end{theorem}
